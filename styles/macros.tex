% ---------------- Packages ----------------
\usepackage{booktabs}
\usepackage{tabularx}
\usepackage{microtype}
\usepackage{amsmath, amssymb}
\usepackage{graphicx}

% ---------------- Fonts (XeLaTeX) ----------------
\usepackage{fontspec}

% Inter (убедись, что установлен в системе)
\setsansfont{Inter}[
  Scale=MatchLowercase,
  UprightFont = *-Regular,
  BoldFont    = *-SemiBold,
  ItalicFont  = *-Italic,
  BoldItalicFont = *-SemiBoldItalic
]
\setmainfont{Inter}
\setmonofont{DejaVu Sans Mono}


% Разрешить beamer использовать fontspec
% \usefonttheme{professionalfonts}

% ---------------- Force sans-serif everywhere ----------------
\AtBeginDocument{%
  % базовый шрифт
  \renewcommand{\familydefault}{\sfdefault}

  % ВСЕ заголовки
  \setbeamerfont{title}{family=\sffamily}
  \setbeamerfont{subtitle}{family=\sffamily}
  \setbeamerfont{frametitle}{family=\sffamily}
  \setbeamerfont{framesubtitle}{family=\sffamily}

  % Структура
  \setbeamerfont{section title}{family=\sffamily}
  \setbeamerfont{subsection title}{family=\sffamily}

  % Footer / header
  \setbeamerfont{author}{family=\sffamily}
  \setbeamerfont{date}{family=\sffamily}
  \setbeamerfont{institute}{family=\sffamily}
}

% ---------------- Beamer UI tweaks ----------------
% Remove navigation icons
\setbeamertemplate{navigation symbols}{}

% Footer with title + slide numbers
\setbeamertemplate{footline}{
  \leavevmode%
  \hbox{%
    \begin{beamercolorbox}[wd=.80\paperwidth,ht=2.6ex,dp=1.2ex,left]{author in head/foot}
      \hspace*{1ex}\insertshorttitle
    \end{beamercolorbox}%
    \begin{beamercolorbox}[wd=.20\paperwidth,ht=2.6ex,dp=1.2ex,right]{date in head/foot}
      \insertframenumber/\inserttotalframenumber\hspace*{1ex}
    \end{beamercolorbox}%
  }%
  \vskip0pt%
}


% --- Smaller bullets for itemize ---
\setbeamertemplate{itemize item}{\scriptsize\raise1.25pt\hbox{$\bullet$}}
\setbeamertemplate{itemize subitem}{\scriptsize\raise1.25pt\hbox{$\bullet$}}
\setbeamertemplate{itemize subsubitem}{\tiny\raise1.25pt\hbox{$\bullet$}}


% Create a section divider slide at each new \section
\AtBeginSection[]{
  \begin{frame}[plain]
    \vfill
    \centering
    {\fontsize{40}{48}\selectfont\insertsection\par}
    \vfill
  \end{frame}
}
